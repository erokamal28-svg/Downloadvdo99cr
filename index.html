function startDownload(quality) { const urlInput = document.getElementById('videoUrl').value.trim(); const status = document.getElementById('status'); const loader = document.getElementById('loader'); if(!urlInput) return alert("Please paste a link first!"); loader.style.display = "block"; status.innerText = "Requesting Server... Please wait."; // আপনার রেন্ডার ইউআরএল const RENDER_URL = "https://downloadvdo99cr.onrender.com"; const finalUrl = `${RENDER_URL}/download?url=${encodeURIComponent(urlInput)}&quality=${quality}`; // সরাসরি উইন্ডো লোকেশন পরিবর্তন করে ডাউনলোড ট্রিগার করা window.location.href = finalUrl; setTimeout(() => { loader.style.display = "none"; status.innerText = "Download requested! Check your phone's notification bar."; }, 5000); }
